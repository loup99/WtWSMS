# "The Marriage Game" - Marry another character.
THE_MARRIAGE_GAME = {
	possible = {
		is_ironman = yes
		NOT = { has_character_flag = achievement_get_married }
	}
	
	happened = {
		has_character_flag = achievement_get_married
	}
}

# "A Pope of My Own" - Set up an antipope.
A_POPE_OF_MY_OWN = {
	possible = {
		is_ironman = yes
		NOT = {
			any_vassal = {
				any_claim = {
					title = k_papal_state
				}
			}
		}
	}

	happened = {
		any_vassal = {
			any_claim = {
				title = k_papal_state
			}
		}
	}
}
