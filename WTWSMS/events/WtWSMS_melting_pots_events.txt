# New Pagan, Tribal and Germanic ruler holding a Christian, sub-Roman and Feudal province, making it a melting-pot target
province_event = {
	id = WtWSMS_pots.1

	hide_window = yes	
	
	only_rulers = yes

	trigger = {
		owner = { 
			religion_group = pagan_group
			is_tribal = yes
			has_germanic_culture_trigger = yes 
		}
		religion_group = christian
		is_feudal = yes
		has_romance_culture_trigger = yes
		NOR = { 
			any_province_holding = {
				has_holding_modifier = recently_conquered
			} 
		}
	}
	
	mean_time_to_happen = {
		years = 1 # Testing placeholder
	}
	
	immediate = {
		set_province_flag = melting_pot_target
	}
}

# Mighty Christian landed vassal brings it up with Pagan liege
character_event = {
	id = WtWSMS_pots.6
	title = WtWSMS_pots.6.name
	desc = WtWSMS_pots.6.desc
	picture = GFX_evt_stone_church
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = WtWSMS_pots.6.a # Embrace the religion
		trigger = {
			NOT = { trait = zealous }
		}
		ROOT = {
			religion = FROM
			religion_authority = {
				modifier = ruler_converted_from
			}
			if = {
				limit = {
					trait = cynical
				}
				random = {
					chance = 60
					add_trait = sympathy_pagans
				}
			}
			if = {
				limit = {
					NOT = { trait = cynical }
				}
				random = {
					chance = 50
					add_trait = sympathy_pagans
				}
			}		
			hidden_tooltip = {
				religion_authority = {
					modifier = ruler_converted_to
				}
			}
			piety = 200
			character_event = {
				id = WtWSMS_pots.17  # Conversion to Christianity
			}
		}
	}
	
	option = {
		name = WtWSMS_pots.6.b # Make vague statement
		character_event = { 
			id = WtWSMS_pots.15 # Insists after first step
			days = 100
		}
	}

	option = {
		name = WtWSMS_pots.6.c # Refuse
		set_character_flag = refused_conversion_christianity
	}
}

	desc = WtWSMS_pots.40.desc
	picture = GFX_evt_stone_church # Placeholder
	border = GFX_event_normal_frame_diplomacy

	trigger = {
		OR = { 
			had_global_flag = { 
				flag = old_french_melting_pot
				days = 3650 # 10 years 
			}
			had_global_flag = { 
				flag = gallo_romance_melting_pot
				days = 3650 # 10 years 
			}
			had_global_flag = { 
				flag = occitan_melting_pot
				days = 3650 # 10 years 
			}
		}
		NOT = { # Lack of strong Romano-Gallic realm remaining
			any_independent_ruler = {
				culture = romano_gallic
				OR = { 
					higher_tier_than = duke
					realm_size = 35
					any_demesne_title = {
						count = 3
						tier = DUKE
					}
				}
			}
		}
		NOT = {
			owner = { 
				OR = { 
					culture = romano_gallic 
					top_liege = {
						culture = romano_gallic
					}
					liege = {
						culture = romano_gallic
					}
				}
			}
		}
		culture = romano_gallic
	}

	mean_time_to_happen = {
		years = 1 # Testing placeholder
	}
	
	option = {
		name = WtWSMS_pots.40.a
		if = {
			limit = {
				owner = {
					culture = old_frankish
				}
				region = custom_frankish
			}
			culture = frankish
		}
		else_if = {
			limit = {
				owner = {
					has_germanic_culture_trigger = yes
					NOT = { culture = old_frankish }
				}
				region = custom_frankish
			}
			culture = gallo_romance
		}
		if = {
			limit = {
				owner = {
					culture = old_frankish
				}
				region = custom_occitan
			}
			culture = occitan
		}
	}
