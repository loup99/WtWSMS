# New Pagan, Tribal and Germanic ruler holding a Christian, sub-Roman and Feudal province, making it a melting-pot target
province_event = {
	id = WtWSMS_pots.1

	hide_window = yes
	
	trigger = {
		has_game_rule = { # Has enabled new melting pot events
			name = melting_pots
			value = new
		}
		owner = {
			OR = {
				religion_group = pagan_group
				religion_group = christian # Can already be Christian at arrival
			}
			OR = {
				is_tribal = yes
				is_feudal = yes # Can already be Feudal at arrival
			}
		}
		OR = {
			owner = { # Germanic
				has_germanic_culture_trigger = yes
			}
			AND = { # Celtic alternative for Romano-Gallic and Romano-British
				owner = {
					has_celtic_culture_trigger = yes
				}
				OR = {
					culture = romano_gallic
					culture = romano_british
				}
			}
			AND = { # Slavic alternative for Thraco-Roman and Rumanescu
				owner = {
					has_slavic_culture_trigger = yes
				}
				OR = {
					culture = thraco_roman
					culture = romano_dacian
				}
			}
			AND = { # Arabic alternative for Andalusian
				owner = {
					has_arabic_culture_trigger = yes
				}
				culture = romano_hispanic
			}
		}
		religion_group = christian
		county = {
			is_feudal = yes
		}
		has_romance_culture_trigger = yes
		NOR = {
			any_province_holding = {
				has_holding_modifier = recently_conquered
			}
		}
	}
	
	mean_time_to_happen = {
		years = 1
	}
	
	immediate = {
		set_province_flag = melting_pot_target
	}
}

# Mighty Christian landed vassal brings it up with Pagan liege
character_event = {
	id = WtWSMS_pots.6
	desc = WtWSMS_pots.6.desc
	picture = GFX_evt_throne_room
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = WtWSMS_pots.6.a # Embrace the religion
		trigger = {
			NOT = { trait = zealous }
		}
		ROOT = {
			religion = FROM
			religion_authority = {
				modifier = ruler_converted_from
			}
			if = {
				limit = {
					trait = cynical
				}
				random = {
					chance = 60
					add_trait = sympathy_pagans
				}
			}
			else = {
				random = {
					chance = 50
					add_trait = sympathy_pagans
				}
			}
			hidden_tooltip = {
				religion_authority = {
					modifier = ruler_converted_to
				}
			}
			piety = 200
			character_event = {
				id = WtWSMS_pots.17 # Conversion to Christianity
			}
		}
		FROM = {
			character_event = {
				id = WtWSMS_pots.125 # Liege converts to Christianity
			}
		}
	}
	
	option = {
		name = WtWSMS_pots.6.b # Make vague statement
		character_event = {
			id = WtWSMS_pots.15 # Insists after first step
			days = 100
		}
	}

	option = {
		name = WtWSMS_pots.6.c # Refuse
		set_character_flag = refused_conversion_christianity
		FROM = {
			character_event = {
				id = WtWSMS_pots.126 # Liege refuses conversion to Christianity
			}
		}
	}
}
