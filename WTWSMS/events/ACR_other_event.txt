#AI allowed to use the decision
character_event = {
	id = GoPagan.120
	
	hide_window = yes
	
	min_age = 16
	capable_only = yes
	prisoner = no
	ai = yes

	trigger = {
		is_heretic = no
		piety = 0
		health = 3
		is_priest = no
		controls_religion = no
		OR = {
			NOT = { religion_authority = 0.20 }
			AND = {
				NOT = { religion_authority = 0.40 }
				trait = lunatic
			}
		}
		NOT = {
			has_character_flag = can_go_pagan
			has_character_flag = gone_pagan
			trait = zealous
			trait = crusader
			trait = mujahid
			trait = holy_warrior
			has_character_modifier = voice_of_jesus
			has_character_modifier = voice_of_satan
			piety = 500
			religion = monophysite
			religion = miaphysite
			religion = pagan
			AND = {
				religion_group = pagan_group
				is_reformed_religion = no
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.25
			trait = lunatic
		}
		modifier = {
			factor = 0.25
			trait = cynical
		}
		modifier = {
			factor = 0.25
			has_character_modifier = defied_church
		}
		modifier = {
			factor = 0.75
			NOT = { religion_authority = 0.15 }
		}
		modifier = {
			factor = 0.5
			NOT = { religion_authority = 0.10 }
		}
		modifier = {
			factor = 0.5
			trait = mystic
		}
		modifier = {
			factor = 0.5
			location = { religion_group = pagan_group }
		}
		modifier = {
			factor = 0.5
			independent = no
			liege = { religion_group = pagan_group }
		}
		modifier = {
			factor = 0.5
			OR = {
				trait = sympathy_pagans
				trait = sympathy_zoroastrianism
				trait = sympathy_judaism
				trait = sympathy_indian
			}
		}
		modifier = {
			factor = 0.75
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 0.75
			NOT = { learning = 2 }
		}
		modifier = {
			factor = 1.5
			piety = 100
		}
		modifier = {
			factor = 1.5
			piety = 200
		}
		modifier = {
			factor = 1.5
			piety = 300
		}
		modifier = {
			factor = 1.5
			piety = 400
		}
		modifier = {
			factor = 3
			trait = slothful
		}
		modifier = {
			factor = 3
			trait = content
		}
		modifier = {
			factor = 3
			trait = craven
		}
		modifier = {
			factor = 3
			trait = patient
		}
		modifier = {
			factor = 3
			learning = 10
		}
		modifier = {
			factor = 3
			learning = 15
		}
		modifier = {
			factor = 3
			trait = scholar
		}
		modifier = {
			factor = 5
			OR = {
				trait = pilgrim
				trait = hajjaj
			}
		}
		modifier = {
			factor = 5
			independent = no
			is_female = yes
			any_spouse = { religion = ROOT }
		}
		modifier = {
			factor = 10

		}
		modifier = {
			factor = 10
			NOT = {
				trait = cynical
				trait = lunatic
				any_spouse = { religion_group = pagan_group }
				any_dynasty_member = { religion_group = pagan_group }
				any_liege = { religion_group = pagan_group }
				any_vassal = { religion_group = pagan_group }
				location = { religion_group = pagan_group }
				location = { any_province_character = { religion_group = pagan_group } }

	immediate = {
		set_character_flag = can_go_pagan

	}

# Ask the player if they want to "paganify" the world
character_event = {
	id = ancrel.9001
	desc = ancrel.9001.desc
	picture = GFX_evt_Pagan_dolmen
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = acr_paganify }
	}

	immediate = {
		set_global_flag = acr_paganify
	}

	option = {
		name = ancrel.9001.a
		any_independent_ruler = {
			any_realm_character = {
				if = { # Celtic
					limit = {
						NOT = { religion_group = pagan_group }
						OR = {
							culture_group = celtic
							culture_group = iberian
							culture_group = west_germanic
							culture = occitan
							culture = frankish
							culture = norman
						}
					}
					character_event = { id = ancrel.9002 }
				}
				if = { # Hellenic
					limit = {
						NOT = { religion_group = pagan_group }					
						OR = {
							culture_group = byzantine
							culture = italian
						}

					}
					character_event = { id = ancrel.9003 }
				}
				if = { # Kemetic
					limit = {
						NOT = { religion_group = pagan_group }					
						OR = {
							culture = bedouin_arabic
							culture = levantine_arabic
							culture = egyptian_arabic
							culture_group = east_african
						}
					}
					character_event = { id = ancrel.9004 }
				}
				if = { # West African
					limit = {
						NOT = { religion_group = pagan_group }					
						OR = {
							culture = maghreb_arabic
							culture = andalusian_arabic
							culture_group = west_african
						}
					}
					character_event = { id = ancrel.9005 }
				}
				if = { # Slavic
					limit = {
						NOT = { religion_group = pagan_group }					
						OR = {
							culture_group = west_slavic
							culture_group = east_slavic
							culture_group = south_slavic
						}
					}
					character_event = { id = ancrel.9006 }
				}
				if = { # Norse
					limit = {
						NOT = { religion_group = pagan_group }					
						OR = {
							culture_group = north_germanic
							culture_group = central_germanic
						}
					}
					character_event = { id = ancrel.9007 }					
				}
				if = { # Zoroastrian
					limit = {
						NOT = { religion_group = pagan_group }					
						culture_group = iranian
					}
					character_event = { id = ancrel.9008 }
				}
				if = { # Tengri
					limit = {
						NOT = { religion_group = pagan_group }
						OR = {
							culture_group = altaic
							culture_group = magyar
						}
					}
					character_event = { id = ancrel.9009 }
				}
			}
		}
	}
}
