# Druid Warns About Child (AI Spouse)
character_event = {
	id = ancrel.0211
	hide_window = yes
	
	only_women = yes
	min_age = 16
	prisoner = no
	
	trigger = {
		has_dlc = "The Old Gods"
		is_pregnant = yes
		spouse = {
			supernatural_events_trigger = yes
			any_courtier = { 
				trait = druid 
			}
		}
		NOT = { has_character_flag = child_stolen_sidhe }
	}
	
	mean_time_to_happen = {
		months = 550
	}
	
	immediate = {
		set_character_flag = child_stolen_sidhe
		spouse = {
			random_courtier = { 
				limit = { trait = druid }
				character_event = { id = ancrel.0212 }
			}
		}
	}
}

character_event = {
	id = ancrel.0222
	hide_window = yes
	
	ai = yes
	
	trigger = {
		has_dlc = "The Old Gods"
		age = 4
		NOT = { age = 16 }
		NOT = { has_character_flag = changeling_child_kills_1 }
		has_character_flag = changeling_child
		father = { is_alive = yes }
		OR = {
			has_character_flag = changeling_child_strange_1
			has_character_flag = changeling_child_strange_2
			has_character_flag = changeling_child_strange_3
		}
		father = {
			any_courtier = {
				NOT = { 
					character = ROOT
					is_close_relative = ROOT 
				}
				host = { 
					character = liege
					prisoner = no
				}
				ai = yes
				is_primary_heir = no
				is_pretender = no
				prisoner = no
				age = 13
				in_command = no
				NOT = { spouse = { character = host } }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 320
	}
	
	immediate = {
		set_character_flag = changeling_child_kills_1
		father = { 
			random_courtier = {
				limit = {
					NOT = { 
						character = ROOT
						is_close_relative = ROOT 
					}
					host = { 
						character = liege
						prisoner = no
					}
					ai = yes
					is_primary_heir = no
					is_pretender = no
					prisoner = no
					age = 13
					in_command = no
					NOT = { spouse = { character = host } }
				}
				character_event = { id = ancrel.0223 } 
			}
		}
	}
}
