# Philosopher approaches the Emperor as the realm crumbles
# [on_yearly_pulse]
character_event = {
	id = ancrel.2000
	desc = ancrel.2000.desc
	picture = GFX_evt_emissary
	border = GFX_event_normal_frame_religion

	min_age = 16
	capable_only = yes
	prisoner = no

	is_triggered_only = yes

	trigger = {
		has_dlc = "The Old Gods"

		OR = {
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
		}

		NOR = {
			realm_size = 101 						# empire is in big trouble, smaller than the alexiad start date
			has_character_flag = ancrel_philosopher # hasn't had the event already
			religion_group = pagan_group 			# is still christian
			any_courtier = {
				has_character_flag = special_priest # don't already have the priest in your court
			}
		}
	}

	immediate = {
		set_character_flag = ancrel_philosopher
	}

	option = { # allow the philosopher to enter the court
		name = ancrel.2000.a
		custom_tooltip = {
			text = ancrel.2000.a.tt
			create_character = {
				random_traits = no
				name = "Michael"
				dynasty = none
				religion = ROOT
				culture = greek
				female = no
				age = 31
				attributes = {
					martial = 4
					diplomacy = 8
					stewardship = 6
					intrigue = 8
					learning = 12
				}
				health = 6
				fertility = 0.5
				trait = mastermind_theologian
				trait = proud
				trait = cynical
				trait = scholar
				trait = temperate
				trait = genius
				trait = gregarious
				trait = stoic # most popular school throughout history
			}
			new_character = {
				opinion = {
					modifier = opinion_patron
					who = ROOT
				}
				give_nickname = nick_the_stammerer
				set_character_flag = special_priest
			}
		}
	}

	option = { # deny the request
		name = ancrel.2000.b
		custom_tooltip = { text = ancrel.2000.b.tt }
	}
}
