# Make Hellenic Paganism the state religion?
# Fired on creation of the Pontificate Maxima for Roman/Latin/Byzantine empire
character_event = {
	id = ancrel.2102
	desc = EVTDESCancrel.2102
	picture = GFX_evt_roman_temple
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes
	hide_from = yes

	option = {
		name = EVTOPTAancrel.2102
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.25
				trait = zealous
			}
			modifier = {
				factor = 0.8
				trait = cynical
			}
			modifier = {
				factor = 1.25
				trait = proud
			}
			modifier = {
				factor = 0.8
				trait = humble
			}
			modifier = {
				factor = 0.8
				OR = {
					trait = mastermind_theologian
					trait = scholarly_theologian
					trait = scholar
					trait = mystic
				}
			}
		}
		piety = 50
		if = {
			limit = { NOT = { trait = zealous } }
			random = {
				chance = 50
				add_trait = zealous
			}
		}
		custom_tooltip = {
			text = EVTTOOLTIPAancrel.2102
			give_nickname = nick_the_apostate
			hidden_tooltip = {
				any_realm_character  = {
					limit = {
						is_adult = yes
						religion = hellenic_pagan
					}
					opinion = { modifier = opinion_hellenic_state_religion who = ROOT }
				}
				any_realm_character  = {
					limit = {
						is_adult = yes
						NOT = { religion_group = pagan_group }
					}
					opinion = { modifier = opinion_forced_hellenic_state_religion who = ROOT }
				}
				capital_scope = {
					religion = ROOT
				}
				any_courtier = {
					limit = { employer = { character = ROOT } }
					religion = ROOT
				}
				any_vassal = {
					limit = { is_playable = no }
					religion = ROOT
				}
				any_vassal = {
					limit = {
						is_playable = yes
						NOT = { religion = ROOT }
					}
					character_event = { id = ancrel.2103 }
				}
			}
		}
	}

	option = {
		name = EVTOPTBancrel.2102
		ai_chance = {
			factor = 50
		}
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		hidden_tooltip = {
			any_realm_character  = {
				limit = {
					is_adult = yes
					religion = hellenic_pagan
				}
				opinion = { modifier = opinion_no_hellenic_state_religion who = ROOT }
			}
			any_realm_character  = {
				limit = {
					is_adult = yes
					NOT = { religion = hellenic_pagan }
				}
				opinion = { modifier = opinion_no_forced_hellenic_state_religion who = ROOT }
			}
		}
	}
}

# Adopt the new Hellenic state religion?
character_event = {
	id = ancrel.2103
	desc = EVTDESCancrel.2103
	picture = GFX_evt_roman_temple
	border = GFX_event_normal_frame_religion

	is_triggered_only = yes

	option = {
		name = EVTOPTAancrel.2103
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0.5
				trait = zealous
			}
			modifier = {
				factor = 2
				trait = cynical
			}
			modifier = {
				factor = 0.8
				trait = proud
			}
			modifier = {
				factor = 1.25
				trait = humble
			}
			modifier = {
				factor = 0.8
				religion_authority = 0.8
			}
			modifier = {
				factor = 0.8
				religion_authority = 0.6
			}
			modifier = {
				factor = 1.25
				NOT = { religion_authority = 0.4 }
			}
			modifier = {
				factor = 1.25
				NOT = { religion_authority = 0.2 }
			}
			modifier = {
				factor = 1.25
				opinion = { who = FROM value = 75 }
			}
			modifier = {
				factor = 1.25
				opinion = { who = FROM value = 50 }
			}
			modifier = {
				factor = 0.8
				NOT = { opinion = { who = FROM value = 25 } }
			}
			modifier = {
				factor = 0.8
				NOT = { opinion = { who = FROM value = 0 } }
			}
			modifier = {
				factor = 0.5
				religion_group = pagan_group
			}
			modifier = {
				factor = 2.0
				OR = {
					culture = roman
					culture = italian
					culture = greek
					culture = sicilian_greek # SWMH
					#culture_group = italian_group # SWMH, doesn't exist in WtWSMS
					culture = latin #WtWSMS
				}
			}
			modifier = {
				factor = 2.0
				liege = { character = FROM }
			}
			modifier = {
				factor = 2.0
				crownlaw_title = {
					OR = {
						# free infidel revokation
						has_law = centralization_2
						has_law = centralization_3
						has_law = centralization_4
					}
				}
			}
		}
		if = {
			limit = { religion_group = christian }
			add_trait = sympathy_christendom
		}
		religion = FROM
		if = {
			limit = { trait = zealous }
			remove_trait = zealous
		}
		hidden_tooltip = {
			capital_scope = {
				religion = FROM
			}
			any_courtier = {
				limit = { employer = { character = ROOT } }
				religion = FROM
			}
			any_vassal = {
				limit = { is_playable = no }
				religion = FROM
			}
			any_vassal = {
				limit = {
					is_playable = yes
					NOT = { religion = FROM }
				}
				character_event = { id = ancrel.2103 }
			}
		}
		reverse_opinion = { modifier = opinion_adopted_hellenic_state_religion who = FROM }
	}

	option = {
		name = EVTOPTBancrel.2103
		ai_chance = { factor = 80 }
		piety = 10
		reverse_opinion = { modifier = opinion_rejected_hellenic_state_religion who = FROM }
	}
}

# Hellenic Legion forms
narrative_event = {
	id = ancrel.2109
	title = EVTTITLEancrel.2109
	desc = EVTDESCancrel.2109
	picture = GFX_evt_crusaders
	border = GFX_event_narrative_frame_religion
	
	major = yes
	only_playable = yes
	
	trigger = {
		religion = hellenic_pagan
		is_heretic = no
		NOT = { is_title_active = d_legio_hellenica }
		OR = {
			e_byzantium = {
				holder_scope = {
					religion = hellenic_pagan
				}
			}
			e_roman_empire = {
				holder_scope = {
					religion = hellenic_pagan
				}
			}
			496 = { # Constantinople
				owner = { religion = hellenic_pagan }
			}
			333 = { # Rome
				owner = { religion = hellenic_pagan }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 4
	}
	
	immediate = {
		activate_title = { title = d_legio_hellenica status = yes }
		create_random_soldier = {
			dynasty = none
			random_traits = yes
			culture = roman
			religion = hellenic_pagan
		}
		new_character = {
			d_legio_hellenica = {
				grant_title = PREV
			}
			set_defacto_liege = THIS
			wealth = 500
		}
		
		religion_authority = {
			modifier = religious_order_formed
			years = 10
		}
	}
	option = {
		name = EVTOPTAancrel.2109
	}
}
