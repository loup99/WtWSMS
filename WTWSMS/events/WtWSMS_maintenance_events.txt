# Maintenance events

namespace = misc_event

# Destroy Bagaude
character_event	= {
	id = misc_event.1
	desc = misc_event.1.desc
	hide_window = yes	
	
	ai = yes
	only_rulers = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		has_landed_title = d_bagaudae
		ROOT = { 
			NOT = { 
				primary_title = {
					title = d_bagaudae
				}
			}
		}
	}
	
	option = {
		name = OK
		
		activate_title	= { 
			title = d_bagaudae 
			status = no 
		}
	}
}

# Maintain Pope Nicene
character_event	= {
	id = misc_event.2
	desc = misc_event.2.desc
	hide_window = yes	
	
	ai = yes
	only_rulers = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		OR = {
			has_landed_title = d_papal_state
			has_landed_title = d_orthodox 
		}
		NOT = { religion = nicene }
	}
	
	option = {
		name = OK
		
		ROOT = {
			abandon_heresy = yes
			religion = nicene 
		}
	}
}

# Maintain Tzadikite and Melchisedechian Toddlers from becoming Mystics when their parents or courtiers receive the Trait
character_event = {
	id = misc_event.5

	hide_window = yes

	mean_time_to_happen = {
		days = 1
	}

	trigger = {
		NOT = { age = 5 }
		trait = mystic
		is_landed = no
	}

	option = {
		name = Okay
		
		ROOT = {
			remove_trait = mystic
		}
	}
}
