# Griot calculator
character_event = {
	id = wafrica.0
	#desc = wafrica.0.desc #not needed if hide_window = yes
	picture = GFX_evt_griot
	
	hide_window = yes
	
	is_triggered_only = yes
	
	only_rulers = yes
	
	trigger = {
		OR = {
			culture_group = west_african
			culture_group = central_african
			religion = west_african_pagan
			religion = west_african_pagan_reformed
		}
		primary_title = { higher_tier_than = BARON }
	}
	
	immediate = {
		if = {
			limit = {
				NOT = {
					any_vassal = {
						has_minor_title = title_court_griot
					}
				}
				NOT = {
					any_courtier = {
						has_minor_title = title_court_griot
					}
				}			
			}
			add_character_modifier = {
				name = VIET_no_griot
				duration = 400
			}
			break = yes
		}
		if = {
			limit = {
				OR = {
					any_vassal = {
						has_minor_title = title_court_griot
					}
					any_courtier = {
						has_minor_title = title_court_griot
					}
				}			
			}
			if = {
				limit = {
					any_vassal = {
						has_minor_title = title_court_griot
					}
				}
				random_vassal = {
					limit = {
						has_minor_title = title_court_griot
					}
					save_event_target_as = VIET_griot_mark
				}
			}
			if = {
				limit = {
					any_courtier = {
						has_minor_title = title_court_griot
					}
				}
				random_courtier = {
					limit = {
						has_minor_title = title_court_griot
					}
					save_event_target_as = VIET_griot_mark
				}
			}
			if = {
				limit = {
					has_character_modifier = VIET_no_griot
				}
				remove_character_modifier = VIET_no_griot
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_warriors_1
				}
				remove_character_modifier = VIET_griot_inspiring_warriors_1
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_warriors_2
				}
				remove_character_modifier = VIET_griot_inspiring_warriors_2
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_warriors_3
				}
				remove_character_modifier = VIET_griot_inspiring_warriors_3
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_community_1
				}
				remove_character_modifier = VIET_griot_inspiring_community_1
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_community_2
				}
				remove_character_modifier = VIET_griot_inspiring_community_2
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_community_3
				}
				remove_character_modifier = VIET_griot_inspiring_community_3
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_spirits_1
				}
				remove_character_modifier = VIET_griot_inspiring_spirits_1
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_spirits_2
				}
				remove_character_modifier = VIET_griot_inspiring_spirits_2
			}
			if = {
				limit = {
					has_character_modifier = VIET_griot_inspiring_spirits_3
				}
				remove_character_modifier = VIET_griot_inspiring_spirits_3
			}
			random_list = {
				33 = { character_event = { id = wafrica.2 } } #warriors
				33 = { character_event = { id = wafrica.3 } } #community
				33 = { character_event = { id = wafrica.4 } } #spirits
			}
		}
	}
}

	id = wafrica.11
	title = wafrica.11.name
	desc = wafrica.11.desc
	picture = GFX_evt_griot
	
	only_rulers = yes
	only_independent = yes

	trigger = {
		AND = {
			culture_group = west_african
			religion_group = muslim
		}
		NOT = { trait = noble_stranger_descendant }
		NOT = { trait = bilal_descendant }
		primary_title = { 
			higher_tier_than = DUKE
			any_previous_holder = { #Will not fire if Kingdom is usurped by someone else or reformed
				NOT = {
					OR = {
						trait = bilal_descendant
						trait = noble_stranger_descendant
					}
				}
			}
		}
		NOT = { has_global_flag = descent_from_bilal_claimed }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_global_flag = descent_from_bilal_claimed
		set_character_flag = le_lion_king
		set_dynasty_flag = descendants_of_bilal
		if = {
			limit = {
				NOT = { has_global_flag = lion_king_nickname_given }
			}
			character_event = { id = wafrica.46 days = 10 random = 10 }
		}
	}
	
	option = {
		name = wafrica.11.a #Ok
		add_trait = bilal_descendant		
		prestige = 100
		piety = 100
		hidden_tooltip = {
			any_dynasty_member = {
				limit = {
					NOT = {
						character = ROOT
					}
				}
				character_event = { id = wafrica.12 }
			}		
		}
	}
}
