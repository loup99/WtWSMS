# You summon your vassals to a Thysia
character_event = {
	id = ancrel.3600
	desc = EVTDESCancrel.3600
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTAancrel.3600
		any_vassal = {
			limit = {
				religion = titan_pagan
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.3601 tooltip = EVTTOOLTIPancrel.3601 }
			hidden_tooltip = { character_event = { id = ancrel.3605 days = 10 } }
		}
		any_vassal = {
			limit = {
				OR = {
					religion_group = christian
					religion_group = muslim
					religion_group = indian_group
					religion = promethean_pagan
					religion = hellenic_pagan
				}
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = ancrel.3602 tooltip = EVTTOOLTIPancrel.3602 }
		}
		hidden_tooltip = { character_event = { id = ancrel.3604 days = 10 } }
	}
}

# Summoned to Thysia (Vassal)
character_event = {
	id = ancrel.3601
	desc = EVTDESCancrel.3601
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTAancrel.3601
		set_character_flag = attending_titan_thysia
		set_character_flag = do_not_disturb

			limit = { trait = in_hiding	}
			remove_trait = in_hiding
			add_character_modifier = {
				name = went_out_of_hiding_timer
				duration = 180
				hidden = yes

		}
		hidden_tooltip = { character_event = { id = ancrel.3699 days = 50 } } # Safety catch flag clearing
	}
}

# Non-Titanic Pagans pay to skip Thysia
character_event = {
	id = ancrel.3602
	desc = EVTDESCancrel.3602
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = { # Pay
		name = EVTOPTAancrel.3602
		trigger = {
			wealth = 5
		}
		FROM = {
			character_event = { id = ancrel.3603 tooltip = EVTTOOLTIPancrel.3603 }
		}
		wealth = -5
	}
	option = { # Don't pay
		name = EVTOPTBancrel.3602
		FROM = {
			character_event = { id = ancrel.3636 tooltip = EVTTOOLTIPancrel.3636 }
		}
		tooltip = {
			FROM = {
				opinion = {
					modifier = opinion_refused_titan_thysia
					who = ROOT
				}
				any_vassal = {
					limit = { has_character_flag = attending_titan_thysia }
					opinion = {
						modifier = opinion_refused_titan_thysia
						who = ROOT
					}
				}
			}
		}
	}
}

					}
