			factor = 0.9
			OR = {
				trait = mastermind_theologian
				trait = scholarly_theologian
				trait = scholar
				trait = mystic
			}
		}
		modifier = {
			factor = 0.9
			OR = {
				trait = zealous
				trait = cynical
			}
		}
	}
	
	immediate = {
		set_character_flag = flag_studied_the_classics
	}
	
	# Decide to emulate one of the epic heroes
	option = {
		name = EVTOPTAancrel.3000
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.75
				NOT = { 
					OR = {
						culture = italian
						culture = roman	
						culture = greek
						culture_group = italian_group # SwMH
						culture_group = sicilian_group # SWMH
					}
				}

				}
			}
			modifier = {
				factor = 1.5
				trait = ambitious
			}

				factor = 1.25

			}

	}
	# Be boring and just let it go
	option = {
		name = EVTOPTBancrel.3000

	}

	picture = GFX_evt_roman_temple

	trigger = {
		has_dlc = "The Old Gods"

		has_character_flag = personal_legend_5
