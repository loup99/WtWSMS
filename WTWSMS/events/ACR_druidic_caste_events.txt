#[on_yearly_pulse] Liege doesn't have a Druid - fired on a yearly basis to remind players
character_event = {
	id = ancrel.0153
	desc = EVTDESCancrel.0153
	picture = GFX_evt_druid
	
	is_triggered_only = yes
	
	only_rulers = yes
	
	trigger = {
		ai = no
		OR = {
			religion = celtic_pagan
			religion = celtic_pagan_reformed
		}
		NOT = {
			OR = {
				any_courtier = {
					has_minor_title = title_court_druid
				}
				any_vassal = {
					has_minor_title = title_court_druid
				}
			}
		}
	}
	
	option = {
		name = EVTOPTBancrel.0153
		trigger = {
			any_courtier = {
				trait = druid
				is_adult = yes
				is_ruler = no # Councillor landed vassals are courtier
				OR = {
					religion = celtic_pagan
					religion = celtic_pagan_reformed
				}
			}
		}
		random_courtier = {
			limit = { 
				trait = druid 
				is_adult = yes
				is_ruler = no # Councillor landed vassals are courtier
				OR = {
					religion = celtic_pagan
					religion = celtic_pagan_reformed
				}
			}
			give_minor_title = title_court_druid
		}
	}

	option = {
		name = EVTOPTAancrel.0153 #Ok
	}
}

# Druid can become known as wicked, if he can't hide his sins
character_event = {
	id = ancrel.0157
	desc = EVTDESCancrel.0157
	picture = GFX_evt_druid
	
	trigger = {
		has_dlc = "Sons of Abraham"
		OR = {
			religion = celtic_pagan
			religion = celtic_pagan_reformed
		}
		trait = druid
		NOT = { trait = bad_druid }
		NOT = { trait = incapable }
		NOT = { intrigue = 3 }
		NOT = { diplomacy = 5 } # 5 instead of 3 (vanilla) to compensate from druid trait +2 bonus
		OR = {
			trait = deceitful 
			trait = cruel
			trait = arbitrary 
			trait = envious
			trait = craven
			trait = slothful
			trait = shy
			trait = paranoid
		}
	}

	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.5
			trait = ambitious
		}
	}
	
	option = {
		name = EVTOPTAancrel.0157
		add_trait = bad_druid
		remove_trait = druid
		if = {
			limit = {
				is_ruler = yes
				has_nickname = no
			}
			give_nickname = nick_the_wicked
		}
	}
}

# [on_yearly_childhood_pulse]
character_event = {
	id = ancrel.0184
	hide_window = yes
	
	is_triggered_only = yes
	prisoner = no

	trigger = {
		age = 14
		OR = {
			religion = celtic_pagan
			religion = celtic_pagan_reformed
		}	
		OR = {
			father_even_if_dead = { is_alive = yes }
			mother_even_if_dead = { is_alive = yes }
		}
		NOR = {
			trait = incapable
			has_character_flag = ancrel_training
		}
	}
	
	immediate = {
		set_character_flag = ancrel_training
		if = {
			limit = {
				father = {
					is_alive = yes
					any_child = {
						character = ROOT
						mother_even_if_dead = {
							OR = {
								is_alive = no
								NOT = { higher_real_tier_than = PREVPREV }
							}
						}
					}
				}
			}
			father = { character_event = { id = ancrel.0183 days = 3 } }
			break = yes
		}
		if = {
			limit = {
				mother = {
					is_alive = yes
					any_child = {
						character = ROOT
						father_even_if_dead = {
							OR = {
								is_alive = no
								NOT = { higher_real_tier_than = PREVPREV }
							}
						}
					}
				}
			}
			mother = { character_event = { id = ancrel.0183 days = 3 } }
			break = yes
		}
	}
}
