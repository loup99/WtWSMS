namespace = hellenic

# Informed about Restoration of Rome (Byzantium / Western Roman Empire / Latin Empire)
narrative_event = {
	id = hellenic.221
	title = EVTNAMELoR.21
	desc = EVTDESCLoR.21
	
	picture = GFX_evt_rome_restored_byzantine
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTALoR.21
		trigger = {
			NOT = { has_landed_title = e_hre }
			NOT = { has_landed_title = e_byzantium }
			NOT = { has_landed_title = e_wre }
			NOT = { has_landed_title = e_latin_empire }
		}
	}
	option = {
		name = EVTOPTBLoR.21
		trigger = {
			OR = {
				has_landed_title = e_hre
				has_landed_title = e_byzantium
				has_landed_title = e_wre
				has_landed_title = e_latin_empire
			}
		}
		prestige = -500
	}
}

# The Restoration of Rome (HRE)
narrative_event = {
	id = hellenic.222
	title = EVTNAMELoR.20
	desc = hellenic.222.desc
	
	picture = GFX_evt_rome_restored_byzantine
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTALoR.20
		set_global_flag = rome_restored
		set_global_flag = rome_restored_by_hre
		e_roman_empire = {
			grant_title = ROOT
			make_primary_title = yes
			copy_title_laws = e_hre
			copy_title_history = e_hre
		}
		add_trait = augustus
		give_nickname = nick_the_glorious
		hidden_tooltip = {
			e_hre = {
				if = {
					limit = { ROOT = { has_landed_title = PREV } }
					any_direct_de_jure_vassal_title = {
						de_jure_liege = e_roman_empire
					}
					destroy_landed_title = THIS
					activate_title = { title = e_hre status = no }
				}
			}
			e_italy = {
				if = {
					limit = { ROOT = { has_landed_title = PREV } }
					any_direct_de_jure_vassal_title = {
						de_jure_liege = e_roman_empire
					}
					destroy_landed_title = THIS
					activate_title = { title = e_italy status = no }
				}
			}
			any_playable_ruler = {
				if = {
					limit = {
						NOT = { character = ROOT }
					}
					narrative_event = { id = hellenic.223 }
				}
			}
		}
	}
}

# The Creation of a new Rome
narrative_event = {
	id = hellenic.224
	title = EVTNAMELoR.20
	desc = hellenic.224.desc
	
	picture = GFX_evt_rome_restored_byzantine
	
	is_triggered_only = yes
	
	option = {
		name = EVTOPTALoR.20
		set_global_flag = rome_restored
		set_global_flag = rome_restored_by_holding_italy
		primary_title = {
			e_roman_empire = {
				grant_title = ROOT
				make_primary_title = yes
				copy_title_laws = PREV
			}
		}
		add_trait = augustus
		give_nickname = nick_the_glorious
		hidden_tooltip = {
			e_byzantium = {
				destroy_landed_title = THIS
				activate_title = { title = e_byzantium status = no }
			}
			e_wre = {
				destroy_landed_title = THIS
				activate_title = { title = e_wre status = no }
			}
			e_latin_empire = {
				destroy_landed_title = THIS
				activate_title = { title = e_latin_empire status = no }
			}
			e_italy = {
				if = {
					limit = { ROOT = { has_landed_title = PREV } }
					any_direct_de_jure_vassal_title = {
						de_jure_liege = e_roman_empire
					}
					destroy_landed_title = THIS
					activate_title = { title = e_italy status = no }
				}
			}
			e_hre = {
				set_name="Germania"
			}
			any_playable_ruler = {
				if = {
					limit = {
						NOT = { character = ROOT }
					}
					narrative_event = { id = hellenic.225 }
				}
			}
		}
	}
}
