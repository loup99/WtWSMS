# Rebirth of Old Britain
narrative_event = {
	id = VIETceltic.0
	desc = VIETceltic.0.desc
	picture = GFX_evt_coming_home
	title = VIETceltic.0.title

 	is_triggered_only = yes

	option = {
		name = VIETceltic.0.a
		add_trait = like_king_arthur
		set_character_flag = VIET_the_second_arthur
		prestige = 300
		piety = 200
		primary_title = {
			 k_britain = {
				grant_title = ROOT
				copy_title_laws = PREV
			}
		}
		give_nickname = nick_the_great
		if = {
			limit = {
				NOT = { has_landed_title = k_britain }
			}
			k_britain = {
				grant_title = ROOT
			}
		}
		k_wales = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_britain
			}
		}
		k_brittany = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_britain
			}
		}
		k_dumnonia = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_britain
			}
		}
		k_england = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = k_britain
			}
		}
		k_britain = {
			de_jure_liege = e_britannia
		}
		if = {
			limit = {
				tier = king
				primary_title = {
					NOT = { title = k_britain }
				}
			}
			k_britain = { make_primary_title = yes }
		}
		if = {
			limit = {
				has_landed_title = k_wales
			}
			k_wales = { destroy_landed_title = THIS }
		}
		if = {
			limit = {
				has_landed_title = k_brittany
			}
			k_brittany = { destroy_landed_title = THIS }
		}
		if = {
			limit = {
				has_landed_title = k_dumnonia
			}
			k_dumnonia = { destroy_landed_title = THIS }
		}
		if = { 
			limit = { 
				has_landed_title = k_england 
			} 
			k_england = { destroy_landed_title = THIS } 
		}		
		hidden_tooltip = {
			narrative_event = { id = VIETceltic.1 }
		}
	}

}

	min_age = 16
	
	is_triggered_only = yes

	trigger = {
		OR = {
			NOT = { has_minor_title = title_knight_of_the_round_table }
			has_landed_title = k_britain
			AND = {
				NOT = { liege = { has_landed_title = k_britain } }
				NOT = { top_liege = { has_landed_title = k_britain } }
			}
			trait = incapable
		}
		trait = knight_of_the_round_table
	}

	option = {
		name = VIETceltic.5.a
		remove_trait = knight_of_the_round_table
		remove_title = title_knight_of_the_round_table
	}
}
