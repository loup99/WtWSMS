	id = VIETmisc.3100
	hide_window = yes
	
	only_rulers = yes
	is_patrician = yes
	
	trigger = {
		any_courtier = {
			dynasty = ROOT
			ai = yes
			age = 18
			NOR = {
				age = 31
				trait = incapable
				trait = celibate
				trait = eunuch
				trait = homosexual
			}
			prisoner = no
			is_female = yes
			is_married = no
			is_betrothed = no
			has_lover = no
			is_primary_heir = no
			is_pretender = no
			ROOT = {
				NOT = {
					any_child = {
						character = PREVPREV
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 250
	}
	
	immediate = {
		random_courtier = {
			limit = {
				dynasty = ROOT
				age = 18
				NOR = {
					age = 31
					trait = incapable
					trait = chaste
				}
				prisoner = no
				is_female = yes
				is_married = no
				is_betrothed = no
				is_primary_heir = no
				is_pretender = no
				ROOT = {
					NOT = {
						any_child = {
							character = PREVPREV
						}
					}
				}
			}
			save_event_target_as = family_member
		}
		random_list = {
			33 = {
				character_event = { id = VIETmisc.3101 }
			}
			33 = {
				modifier = {
					factor = 0
					event_target:family_member = {
						NOT = {
							age = 19
						}
					}
				}
				character_event = { id = VIETmisc.3102 }
			}
			33 = {
				modifier = {
					factor = 0
					event_target:family_member = {
						NOT = {
							age = 20
						}
					}
				}
				character_event = { id = VIETmisc.3103 }
