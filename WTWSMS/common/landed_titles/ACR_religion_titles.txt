d_hellenic_pagan = {
	color={ 155 155 235 }
	color2={ 220 220 0 }
	
	capital = 333 #Roma
	
	title = PONTIFEX_MAXIMUS
	title_female = PONTIFICA_MAXIMUS
	#foa = PONTIFEX_MAXIMUS_FOA # Roman Emperor name workaround
	
	creation_requires_capital = yes
	
	allow = {
		always = no
	}
	
	gain_effect = {
		hidden_tooltip = {
			e_roman_empire = {
				owner = {
					if = {
						limit = {
							religion = hellenic_pagan
						}
						character_event = { id = ancrel.2102 days = 1 }
					}
				}
			}
			e_byzantium = {
				owner = {
					if = {
						limit = {
							religion = hellenic_pagan
							NOT = { has_landed_title = e_roman_empire }
						}
						character_event = { id = ancrel.2102 days = 1 }
					}
				}
			}
			e_latin_empire = {
				owner = {
					if = {
						limit = {
							religion = hellenic_pagan
							NOT = { has_landed_title = e_roman_empire }
							NOT = { has_landed_title = e_byzantium }
						}
						character_event = { id = ancrel.2102 days = 1 }
					}
				}
			}
		}
	}
	
	short_name = yes
	
	primary = no
	
	dignity = 100 # Counted as having this many more counties than it does
	
	# Controls a religion
	controls_religion = hellenic_pagan
	
	religion = hellenic_pagan
	
	dynasty_title_names = no 	# Will not be named "Seljuk", etc.
}
