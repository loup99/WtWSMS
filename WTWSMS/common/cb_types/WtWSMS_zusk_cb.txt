migration = {
	name = CB_NAME_MIGRATION
	war_name = WAR_NAME_MIGRATION
	sprite = 8
	truce_days = 3650
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING

	allow_distant = yes

	on_add = {
		#Sets the flag for giving units & giving the provinces away if you win the war
		ROOT = { character_event = { id = zusk_event.12 } }
		#Then gives event troops for every owned province
		ROOT = { character_event = { id = zusk_event.13 } }
		ROOT = { 				
			add_character_modifier = { 
				name = launched_migration_war
				days = 3650
				hidden = yes
			}
   		}
	}

	can_use = {	
		NOT = { year = 800  } #The migration period ended around 800
		ROOT = {
			OR = {
				religion = tengri_pagan
				religion = scythian_pagan
				religion = slavic_pagan
				religion = germanic_pagan
				culture = alan
			}
			NOT = { has_character_modifier = launched_migration_war }
			higher_tier_than = COUNT #Dukes or Kings
			NOT = { culture = FROM }
			independent = yes #No vassals
			war = no #Ensures that the events activate correctly  
		}
		FROM = {
			independent = yes
			higher_tier_than = COUNT	
		}
	}
	is_valid_title = {
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_success = {
		ROOT = {
			prestige = 200
			occupy_minors_of_occupied_settlements = FROM
			gain_all_occupied_titles = { who = FROM type = invasion }
			any_vassal = { 
				limit = {
					higher_tier_than = BARON
				}
				set_defacto_liege = THIS
			}
		}
		FROM = {
			prestige = -200
		}
	}

	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title_destroy_duchies = {
						title = PREV
						enemy = FROM
						is_crusade = yes # Even if the title holder is not participating in the war, gain holdings occupied by all Crusade participants
						type = invasion
					}
				}
			}
		}

		# Usurp the target kingdom itself, if the enemy holds it
		if = {
			limit = {
				holder_scope = {
					character = FROM
				}
			}
			ROOT = { 
				usurp_title = { target = PREV type = invasion } 
			}
		}
		
		# Take the enemy's duchies within the target kingdom
		FROM = {
			any_demesne_title = {
				limit = {
					tier = DUKE
					OR = {
						title = PREVPREV
						de_jure_liege_or_above = PREVPREV
					}
				}
				ROOT = { 
					usurp_title = { target = PREV type = invasion }
				}
			}
		}


		ROOT = { 
			character_event = { id = zusk_event.14 } 
		}
	}

	on_fail = {
		ROOT = {
			prestige = -50
			
			#Does not give away provinces but cleans up the migration_give_away flag used in the event
			character_event = { id = zusk_event.15 }
		}
		FROM = {
			prestige = 50
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 8.0
			}
			
			#Disbands all troops gained in the Migration CB Migration 
			character_event = { id = zusk_event.11 }

			#Does not give away provinces but cleans up the migration_give_away flag used in the event
	   		character_event = { id = zusk_event.15 }
		}
		FROM = {
			prestige = 200
		}
	}

	on_attacker_leader_death = {
		#Does not give away provinces but cleans up the migration_give_away flag used in the event
		ROOT = { character_event = { id = zusk_event.15 } }

		hidden_tooltip = {
			any_defender = {
				letter_event = { id = 251 }
			}
		}

		end_war = whitepeace
	}

	attacker_ai_victory_worth = {
		factor = 150
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = { 
		factor = 1
		modifier = {
			factor = 1.2
			ROOT = {
				religion = tengri_pagan
			}
		}	
		modifier = {
			factor = 0
			OR = {
				 FROM = { culture_group = north_germanic }
				 FROM = { culture_group = finno_ugric }
			 }
		}	
		modifier = {
			factor = 0
						 NOT = { FROM = { realm_size = 10 } }
		}	
		modifier = {
			factor = 0
			ROOT = { realm_size = 30 }
		}		 
		modifier = {
			factor = 0
		 	ROOT = { higher_tier_than = KING }
		}		 
		modifier = {
			factor = 0
			ROOT = { 
				OR = { 
					culture = old_frankish
					culture = old_saxon
					culture = thuringian
					culture = alamannian
				}
			}
		}		   
		modifier = {
			factor = 10
			ROOT = {
				OR = {
					culture = avar
					culture_group = magyar
				}
			}
			FROM = {
				culture = gepid
			}
		}		
	}		
}


unification = {
	name = CB_NAME_UNIFICATION
	war_name = WAR_NAME_INVASION
	sprite = 8
	truce_days = 3650
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = EMPEROR

	can_use_title = {
		has_holder = no
		ROOT = { has_claim = PREV  }
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	is_valid_title = {
		has_holder = no
		ROOT = { 
			has_claim = PREV
		}
		FROM = {
			any_realm_title = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_success = {
		hidden_tooltip = {
			FROM = { #Julis Nepos becomes independant following Odoacers war
				any_vassal = { 
					limit = { character = 175110 } 
					set_defacto_liege = THIS
				}
			}
			ROOT = { #Without this Theodoric doesnt get Italy if he wins
				if = {
					limit = {
						has_character_flag = give_italy_if_win 
						NOT = { has_landed_title = k_italy }
					}
					primary_title = {
						k_italy = {
							grant_title = ROOT
							copy_title_laws = PREV
						}
					}
					clr_character_flag = give_italy_if_win
				}
			}
		}
					
		ROOT = {
			prestige = 200
		}
		FROM = {
			prestige = -200
		}
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 200
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		hidden_tooltip = {
			ROOT = {
				occupy_minors_of_occupied_settlements = FROM
				gain_all_occupied_titles = FROM
			}
		}
	}
	
	on_success_title = {

		# Vassalize the rest
		custom_tooltip = {
			text = pagan_subjugation_tip
			hidden_tooltip = {
				ROOT = {
					subjugate_or_take_under_title = { # If the target only has territory within the kingdom, he is simply vassalized
						title = PREV
						enemy = FROM
					}
				}
			}
		}

		# Vassalize the rest
		custom_tooltip = {
			text = pagan_subjugation_tip
			hidden_tooltip = {
				ROOT = {
					subjugate_or_take_under_title = { # If the target only has territory within the kingdom, he is simply vassalized
						title = PREV
						enemy = FROM
					}
				}
			}
		}
	}

	on_fail = {
		hidden_tooltip = {
			ROOT = {
				clr_character_flag = give_italy_if_win
			}
		}
		FROM = {
			prestige = 100
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 100
		}
		any_defender = {
			limit = { 
				NOT = { 
					character = FROM
				}
			}
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		ROOT = {
			prestige = -200
		}
	}

	on_reverse_demand = {
		ROOT = {
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		FROM = {
			prestige = 200
		}
		FROM = {
			if = {
				limit = { 
					has_dlc = "Legacy of Rome"
					OR = { 
						has_landed_title = e_byzantium
						has_landed_title = e_roman_empire
					}
					religion_group = christian
				}
				hidden_tooltip = { character_event = { id = LoR.30 days = 12 } }
			}
		}
		any_defender = {
			limit = { character = FROM }
			participation_scaled_prestige = 200
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
	}

	attacker_ai_victory_worth = {
		factor = 200
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = { 
		factor = 10
		modifier = {
			factor = 50
			ROOT = {
				character = 63
			}
			FROM = {
				character = 175110
			}
		}	

		modifier = {
			factor = 100
			ROOT = {
				character = 63
			}
			FROM = {
				character = 55
			}
		}
	}
}
