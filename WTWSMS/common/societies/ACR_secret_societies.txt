secret_religious_society_celtic_pagan = {
	primary_attribute = diplomacy
	is_religious = yes
	is_secret = yes
	opinion_to_other_members = +10

	criminal = yes
	sound = secret_cults_interface
	society_ranks_gfx = secret_ranks
	non_interference = {
		FROM = {
			same_realm = ROOT
		}
	}
	
	can_join_society = {
		hidden_trigger = {
			NAND = {
				ai = yes
				is_society_rank_full = { 
					society = secret_religious_society_celtic_pagan
					rank = 1
				}
			}
		}
		age = 16
		any_society_member = { society_member_of = secret_religious_society_celtic_pagan }
		trait = secretly_celtic_pagan
	}

	show_society = {
		any_society_member = { society_member_of = secret_religious_society_celtic_pagan }
		trait = secretly_celtic_pagan
	}
	
	potential = {
		age = 16
		trait = secretly_celtic_pagan
	}
	
	society_rank = {
		level = 1
		limit = 200
		startup_limit = 0
		decisions = {
			secret_religions_induct_child
			secret_religions_evoke_sympathy
		}
	}
	
	society_rank = {
		level = 2
		limit = 100
		startup_limit = 0
		modifier = { 
			is_visible = {
				OR = {
					society_member_of = secret_religious_society_celtic_pagan

				}
			}

		}
